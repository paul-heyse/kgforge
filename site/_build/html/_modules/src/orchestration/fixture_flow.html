107def t_register_in_duckdb(
108    db_path: str,
109    chunks_info: tuple[str, int],
110    dense_info: tuple[str, int],
111    sparse_info: tuple[str, int],
112) -> dict[str, list[str]]:
113    """T register in duckdb.
114
115    Args:
116        db_path (str): TODO.
117        chunks_info: TODO.
118        dense_info: TODO.
119        sparse_info: TODO.
120
121    Returns:
122        dict: TODO.
123    """
124    reg = DuckDBRegistryHelper(db_path)
125    dense_run = reg.new_run("dense_embed", "Qwen3-Embedding-4B", "main", {"dim": 2560})
126    sparse_run = reg.new_run("splade_encode", "SPLADE-v3-distilbert", "main", {"topk": 256})
127    ds_chunks = reg.begin_dataset("chunks", dense_run)
128    reg.commit_dataset(ds_chunks, chunks_info[0], rows=chunks_info[1])
129    ds_dense = reg.begin_dataset("dense", dense_run)
130    reg.commit_dataset(ds_dense, dense_info[0], rows=dense_info[1])
131    ds_sparse = reg.begin_dataset("sparse", sparse_run)
132    reg.commit_dataset(ds_sparse, sparse_info[0], rows=sparse_info[1])
133    reg.register_documents(
134        [
135            Doc(
136                id="urn:doc:fixture:0001",
137                title="Fixture Doc",
138                authors=[],
139                pdf_uri="/dev/null",
140                source="fixture",
141                openalex_id=None,
142                doi=None,
143                arxiv_id=None,
144                pmcid=None,
145                pub_date=None,
146                license="CC0",
147                language="en",
148                content_hash="",
149            )
150        ]
151    )
152    reg.close_run(dense_run, True)
153    reg.close_run(sparse_run, True)
154    return {"runs": [dense_run, sparse_run]}

158def fixture_pipeline(
159    root: str = "/data", db_path: str = "/data/catalog/catalog.duckdb"
160) -> dict[str, list[str]]:
161    """Fixture pipeline.
162
163    Args:
164        root (str): TODO.
165        db_path (str): TODO.
166    """
167    t_prepare_dirs(root)
168    chunks_info = t_write_fixture_chunks(f"{root}/parquet/chunks")
169    dense_info = t_write_fixture_dense(f"{root}/parquet/dense")
170    sparse_info = t_write_fixture_splade(f"{root}/parquet/sparse")
171    summary = t_register_in_duckdb(db_path, chunks_info, dense_info, sparse_info)
172    return summary
