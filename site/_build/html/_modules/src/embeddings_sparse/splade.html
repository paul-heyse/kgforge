 88    def build(self, docs_iterable: Iterable[tuple[str, dict[str, str]]]) -> None:
 89        """Build.
 90
 91        Args:
 92            docs_iterable (Iterable[Tuple[str, Dict]]): TODO.
 93
 94        Returns:
 95            None: TODO.
 96        """
 97        os.makedirs(self.index_dir, exist_ok=True)
 98        df: dict[str, int] = defaultdict(int)
 99        postings: dict[str, dict[str, float]] = defaultdict(lambda: defaultdict(float))
100        N = 0
101        for doc_id, fields in docs_iterable:
102            text = " ".join(
103                [fields.get("title", ""), fields.get("section", ""), fields.get("body", "")]
104            )
105            toks = self._tokenize(text)
106            N += 1
107            c = Counter(toks)
108            for tok, tf in c.items():
109                df[tok] += 1
110                postings[tok][doc_id] = math.log1p(tf)
111        # compute idf and impact weights
112        self.N = N
113        self.df = dict(df)
114        self.postings = {
115            tok: {
116                doc: w * math.log((N - df[tok] + 0.5) / (df[tok] + 0.5) + 1.0)
117                for doc, w in docs.items()
118            }
119            for tok, docs in postings.items()
120        }
121        with open(os.path.join(self.index_dir, "impact.pkl"), "wb") as f:
122            pickle.dump(
123                {"df": self.df, "N": self.N, "postings": self.postings},
124                f,
125                protocol=pickle.HIGHEST_PROTOCOL,
126            )

140    def search(self, query: str, k: int) -> list[tuple[str, float]]:
141        """Search.
142
143        Args:
144            query (str): TODO.
145            k (int): TODO.
146
147        Returns:
148            List[Tuple[str, float]]: TODO.
149        """
150        toks = self._tokenize(query)
151        scores: dict[str, float] = defaultdict(float)
152        for t in toks:
153            posts = self.postings.get(t)
154            if not posts:
155                continue
156            for doc, w in posts.items():
157                scores[doc] += w
158        return sorted(scores.items(), key=lambda x: x[1], reverse=True)[:k]
