124    def _load_from_duckdb(self) -> None:
125        """Compute load from duckdb.
126
127        Carry out the load from duckdb operation.
128        """
129        if not Path(self.db_path).exists():
130            return
131        con = duckdb.connect(self.db_path)
132        try:
133            dataset = con.execute(
134                """
135              SELECT parquet_root FROM datasets
136              WHERE kind='chunks'
137              ORDER BY created_at DESC
138              LIMIT 1
139            """
140            ).fetchone()
141            if not dataset:
142                return
143            root = dataset[0]
144            rows = con.execute(
145                f"""
146                SELECT c.chunk_id, c.doc_id, coalesce(c.section,''), c.text,
147                       coalesce(d.title,'') AS title
148                FROM read_parquet('{root}/*/*.parquet', union_by_name=true) AS c
149                LEFT JOIN documents d ON c.doc_id = d.doc_id
150            """
151            ).fetchall()
152        finally:
153            con.close()
154
155        for chunk_id, doc_id, section, text, title in rows:
156            self.docs.append(
157                FixtureDoc(
158                    chunk_id=chunk_id,
159                    doc_id=doc_id or "urn:doc:fixture",
160                    title=title or "Fixture",
161                    section=section or "",
162                    text=text or "",
163                )
164            )
165
166        self._build_lex()

168    def _build_lex(self) -> None:
169        """Compute build lex.
170
171        Carry out the build lex operation.
172        """
173        self.tf.clear()
174        self.df.clear()
175        for doc in self.docs:
176            tokens = tokenize(doc.text)
177            tf_counts: dict[str, int] = {}
178            for token in tokens:
179                tf_counts[token] = tf_counts.get(token, 0) + 1
180            self.tf.append(tf_counts)
181            for token in set(tokens):
182                self.df[token] = self.df.get(token, 0) + 1
183        self.N = len(self.docs)

185    def search(self, query: str, k: int = 10) -> list[tuple[int, float]]:
186        """Compute search.
187
188        Carry out the search operation for the surrounding component. Generated documentation highlights how this helper collaborates with neighbouring utilities. Callers rely on the routine to remain stable across releases.
189        
190        Parameters
191        ----------
192        query : str
193            Description for ``query``.
194        k : int | None
195            Optional parameter default ``10``. Description for ``k``.
196        
197        Returns
198        -------
199        List[Tuple[int, float]]
200            Description of return value.
201        
202        Examples
203        --------
204        >>> from search_api.fixture_index import search
205        >>> result = search(...)
206        >>> result  # doctest: +ELLIPSIS
207        ...
208        """
209
210        if getattr(self, "N", 0) == 0:
211            return []
212        qtoks = tokenize(query)
213        if not qtoks:
214            return []
215        scores = [0.0] * self.N
216        for i, tf in enumerate(self.tf):
217            score = 0.0
218            for token in qtoks:
219                if token not in self.df:
220                    continue
221                idf = math.log((self.N + 1) / (self.df[token] + 0.5) + 1.0)
222                score += idf * tf.get(token, 0)
223            scores[i] = score
224        ranked = sorted(enumerate(scores), key=lambda item: item[1], reverse=True)
225        return [(index, score) for index, score in ranked[:k] if score > 0.0]
