190def rrf_fuse(lists: list[list[tuple[str, float]]], k_rrf: int) -> dict[str, float]:
191    """Compute rrf fuse.
192
193    Carry out the rrf fuse operation for the surrounding component. Generated documentation highlights how this helper collaborates with neighbouring utilities. Callers rely on the routine to remain stable across releases.
194    
195    Parameters
196    ----------
197    lists : List[List[Tuple[str, float]]]
198        Description for ``lists``.
199    k_rrf : int
200        Description for ``k_rrf``.
201    
202    Returns
203    -------
204    collections.abc.Mapping
205        Description of return value.
206    
207    Examples
208    --------
209    >>> from search_api.app import rrf_fuse
210    >>> result = rrf_fuse(..., ...)
211    >>> result  # doctest: +ELLIPSIS

213    """
214    
215    scores: dict[str, float] = {}
216    for hits in lists:
217        for rank, (doc_id, _score) in enumerate(hits, start=1):

219    return scores

223def apply_kg_boosts(
224    cands: dict[str, float],
225    query: str,
226    direct: float = 0.08,
227    one_hop: float = 0.04,
228) -> dict[str, float]:
229    """Compute apply kg boosts.
230
231    Carry out the apply kg boosts operation for the surrounding component. Generated documentation highlights how this helper collaborates with neighbouring utilities. Callers rely on the routine to remain stable across releases.
232    
233    Parameters
234    ----------
235    cands : collections.abc.Mapping
236        Description for ``cands``.
237    query : str
238        Description for ``query``.
239    direct : float | None
240        Optional parameter default ``0.08``. Description for ``direct``.
241    one_hop : float | None
242        Optional parameter default ``0.04``. Description for ``one_hop``.
243    
244    Returns
245    -------
246    collections.abc.Mapping
247        Description of return value.
248    
249    Examples
250    --------
251    >>> from search_api.app import apply_kg_boosts
252    >>> result = apply_kg_boosts(..., ...)
253    >>> result  # doctest: +ELLIPSIS
254    ...
255    """
256    
257    q_concepts = set()
258    for w in query.lower().split():
259        if w.startswith("concept"):
260            q_concepts.add(f"C:{w.replace('concept', '')}")
261    out = dict(cands)
262    for chunk_id, base in cands.items():
263        linked = set(kg.linked_concepts(chunk_id))
264        boost = 0.0
265        if linked & q_concepts:
266            boost += direct
267        else:
268            for c in linked:
269                if set(kg.one_hop(c)) & q_concepts:
270                    boost += one_hop
271                    break
272        out[chunk_id] = base + boost
273    return out

278def search(req: SearchRequest, _: None = Depends(auth)) -> dict[str, Any]:
279    """Compute search.
280
281    Carry out the search operation for the surrounding component. Generated documentation highlights how this helper collaborates with neighbouring utilities. Callers rely on the routine to remain stable across releases.
282    
283    Parameters
284    ----------
285    req : src.search_api.schemas.SearchRequest
286        Description for ``req``.
287    _ : None | None
288        Optional parameter default ``Depends(auth)``. Description for ``_``.
289    
290    Returns
291    -------
292    collections.abc.Mapping
293        Description of return value.
294    
295    Examples
296    --------
297    >>> from search_api.app import search
298    >>> result = search(...)
299    >>> result  # doctest: +ELLIPSIS
300    ...
301    """
302    
303    # Retrieve from each channel
304    # We don't have a query embedder here; fallback to empty or demo vector
305    dense_hits: list[tuple[str, float]] = []
306    # sparse via BM25 (preferred) and SPLADE
307    bm25_hits: list[tuple[str, float]] = []
308    if bm25:
309        try:
310            bm25_hits = bm25.search(req.query, k=CFG["search"]["sparse_candidates"])
311        except Exception as exc:  # pragma: no cover - defensive fallback for missing indices
312            logger.warning("BM25 search failed, falling back to empty results: %s", exc)
313            bm25_hits = []
314    try:
315        splade_hits = (
316            splade.search(req.query, k=CFG["search"]["sparse_candidates"]) if splade else []
317        )
318    except Exception:
319        splade_hits = []
320
321    # RRF fusion
322    fused = rrf_fuse([dense_hits, bm25_hits, splade_hits], k_rrf=int(CFG["search"]["rrf_k"]))
323    # KG boosts
324    boosted = apply_kg_boosts(
325        fused,
326        req.query,
327        direct=CFG["search"]["kg_boosts"]["direct"],
328        one_hop=CFG["search"]["kg_boosts"]["one_hop"],
329    )
330    # Rank and craft results
331    top = sorted(boosted.items(), key=lambda x: x[1], reverse=True)[: req.k]
332    results: list[dict[str, Any]] = []
333    for chunk_id, score in top:
334        # In real system we'd hydrate title/section via DuckDB; here we echo ids
335        results.append(
336            SearchResult(
337                doc_id=f"doc-of-{chunk_id}",
338                chunk_id=chunk_id,
339                title=f"Title for {chunk_id}",
340                section="Methods",
341                score=float(score),
342                signals={
343                    "rrf": float(fused.get(chunk_id, 0.0)),
344                    "kg_boost": float(boosted[chunk_id] - fused.get(chunk_id, 0.0)),
345                },
346                spans={"start_char": 0, "end_char": 50},
347                concepts=[
348                    {
349                        "concept_id": c,
350                        "label": c,
351                        "match": ("direct" if c in req.query else "nearby"),
352                    }
353                    for c in kg.linked_concepts(chunk_id)
354                ],
355            ).model_dump()
356        )
357    return {"results": results}
