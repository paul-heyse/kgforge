 89def rrf_fuse(lists: List[List[Tuple[str, float]]], k_rrf: int) -> Dict[str, float]:
 90    """Rrf fuse.
 91
 92    Args:
 93        lists (List[List[Tuple[str, float]]]): TODO.
 94        k_rrf (int): TODO.
 95
 96    Returns:
 97        Dict[str, float]: TODO.
 98    """
 99    scores: Dict[str, float] = {}
100    for hits in lists:
101        for rank, (doc_id, _score) in enumerate(hits, start=1):
102            scores[doc_id] = scores.get(doc_id, 0.0) + 1.0 / (k_rrf + rank)
103    return scores

122    # toy: map words 'concept42' to concept id
123    q_concepts = set()

142def search(req: SearchRequest, _: None = Depends(auth)) -> dict:
143    """Search.
144
145    Args:
146        req (SearchRequest): TODO.
147        _: TODO.
148    """
149    # Retrieve from each channel
150    dense_hits: List[Tuple[str, float]] = []  # we don't have a query embedder here; fallback to empty or demo vector
151    # sparse via BM25 (preferred) and SPLADE
152    bm25_hits = bm25.search(req.query, k=CFG["search"]["sparse_candidates"]) if bm25 else []
153    try:
154        splade_hits = splade.search(req.query, k=CFG["search"]["sparse_candidates"]) if splade else []
155    except Exception:
156        splade_hits = []
157
158    # RRF fusion
159    fused = rrf_fuse([dense_hits, bm25_hits, splade_hits], k_rrf=int(CFG["search"]["rrf_k"]))
160    # KG boosts
161    boosted = apply_kg_boosts(fused, req.query,
162                              direct=CFG["search"]["kg_boosts"]["direct"],
163                              one_hop=CFG["search"]["kg_boosts"]["one_hop"])
164    # Rank and craft results
165    top = sorted(boosted.items(), key=lambda x: x[1], reverse=True)[:req.k]
166    results = []
167    for chunk_id, score in top:
168        # In real system we'd hydrate title/section via DuckDB; here we echo ids
169        results.append(SearchResult(
170            doc_id=f"doc-of-{chunk_id}",
171            chunk_id=chunk_id,
172            title=f"Title for {chunk_id}",
173            section="Methods",
174            score=float(score),
175            signals={"rrf": float(fused.get(chunk_id, 0.0)), "kg_boost": float(boosted[chunk_id]-fused.get(chunk_id,0.0))},
176            spans={"start_char": 0, "end_char": 50},
177            concepts=[{"concept_id": c, "label": c, "match": ("direct" if c in req.query else "nearby")} for c in kg.linked_concepts(chunk_id)]
178        ).model_dump())
179    return {"results": results}
