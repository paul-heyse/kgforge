    def detect_query_concepts(query: str) -> list[str]:
    """Detect query concepts.

    Parameters
    ----------
    query : str
        TODO.

    Returns
    -------
    List[str]
        TODO.
    """
    q = query.lower()
    hits = []
    for cid, meta in _CONCEPTS.items():
        if any(kw in q for kw in meta["keywords"]):
